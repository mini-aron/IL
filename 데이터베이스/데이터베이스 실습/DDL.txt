create table customer {
  custid  char(20) not null primary key,
  custname char(10) not null,
  age number,
  grade char(10) default 'bronze',
  job char(10),
  point int default 0
};


create table product{
  proid char(10) null primary key,
  proname char(20) not null,
  stock number,
  price int not null,
  productor char(10)



}

create table order(
  orderid char(10) not nullprimary key,
   custid char(20) ,
  proid char(10) ,
  cnt  int default 0, 
  addr char(30) not null,
  orderdate date,
  foreign key(custid) references costomer(custid),
  foreign key(proid) references product(proid)
)


create table quick(
  quickid char(10),
  quickname char(20),
  address char(30),
  tel char(11),
  primary key(quickid)
);
